<?php
$data = [
    [
        'id' => 1,
        'name' => 'aaa',
    ],
    [
        'id' => 2,
        'name' => 'bbb',
    ],
];

$id_list = array_column($data, 'id');
$name_list = array_column($data, 'name');

var_dump($id_list);
// array(2) {
//   [0]=>
//   int(1)
//   [1]=>
//   int(2)
// }

var_dump($name_list);
// array(2) {
//   [0]=>
//   string(3) "aaa"
//   [1]=>
//   string(3) "bbb"
// }

<?php
$data = [
    [
        'id' => 1,
        'name' => 'aaa',
    ],
    [
        'id' => 2,
        'name' => 'bbb',
    ],
];

$transposed_matrix = [];
foreach ($data[0] as $key => $value) {
    $transposed_matrix[$key] = array_column($data, $key);
}

var_dump($transposed_matrix);
// array(2) {
//     ["id"]=>
